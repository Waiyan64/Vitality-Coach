CREATE TABLE users (user_id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(50), age INT);
CREATE TABLE dietary_restrictions (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(50));
CREATE TABLE physical_limitations (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(50));
CREATE TABLE objectives (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(50));
CREATE TABLE dishes (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(50), place_in_meal ENUM('breakfast dish', 'main dish', 'dessert', 'starter', 'snack', 'main course'), instructions VARCHAR(2000), video_link VARCHAR(100), ingredients VARCHAR(500));
CREATE TABLE exercises (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(50), position_in_workout ENUM('warmup', 'cooldown', 'middle'), sets INT, reps_or_time VARCHAR(20), instructions VARCHAR(1000), video_link VARCHAR(100)); 
CREATE TABLE workouts (id INT AUTO_INCREMENT PRIMARY KEY, day_of_week ENUM('Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday'), user_id INT, FOREIGN KEY (user_id) REFERENCES users(user_id));
CREATE TABLE meals (id INT AUTO_INCREMENT PRIMARY KEY, meal_of_day ENUM('Breakfast', 'Lunch', 'Dinner/Supper', 'Mid-morning Snack', 'Mid-afternoon Snack'),  day_of_week ENUM('Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday'), user_id INT, FOREIGN KEY (user_id) REFERENCES users(user_id));

Bridge Tables:
CREATE TABLE user_dietary_restrictions( user_id INT, dietary_restriction_id INT, PRIMARY KEY (user_id, dietary_restriction_id), FOREIGN KEY (user_id) REFERENCES users(user_id), FOREIGN KEY (dietary_restriction_id) REFERENCES dietary_restrictions(id));
CREATE TABLE user_physical_limitations( user_id INT, physical_limitation_id INT, PRIMARY KEY (user_id, physical_limitation_id), FOREIGN KEY (user_id) REFERENCES users(user_id), FOREIGN KEY (physical_limitation_id) REFERENCES physical_limitations(id));
CREATE TABLE dish_dietary_restrictions(dish_id INT, dietary_restriction_id INT, PRIMARY KEY (dish_id, dietary_restriction_id), FOREIGN KEY (dish_id) REFERENCES dishes(id), FOREIGN KEY (dietary_restriction_id) REFERENCES dietary_restrictions(id));
CREATE TABLE exercise_physical_limitations( exercise_id INT, physical_limitation_id INT, PRIMARY KEY (exercise_id, physical_limitation_id), FOREIGN KEY (exercise_id) REFERENCES exercises(id), FOREIGN KEY (physical_limitation_id) REFERENCES physical_limitations(id));
CREATE TABLE user_objectives(user_id INT, objective_id INT, PRIMARY KEY (user_id, objective_id), FOREIGN KEY (user_id) REFERENCES users(user_id), FOREIGN KEY (objective_id) REFERENCES objectives(id)); 
CREATE TABLE dish_objectives(dish_id INT, objective_id INT, PRIMARY KEY (dish_id, objective_id), FOREIGN KEY (dish_id) REFERENCES dishes(id), FOREIGN KEY (objective_id) REFERENCES objectives(id));
CREATE TABLE exercise_objectives(exercise_id INT, objective_id INT, PRIMARY KEY (exercise_id, objective_id), FOREIGN KEY (exercise_id) REFERENCES exercises(id), FOREIGN KEY (objective_id) REFERENCES objectives(id));
CREATE TABLE workout_exercises(workout_id INT, exercise_id INT, PRIMARY KEY (workout_id, exercise_id), FOREIGN KEY (workout_id) REFERENCES workouts(id), FOREIGN KEY (exercise_id) REFERENCES exercises(id));
CREATE TABLE meal_dishes(meal_id INT, dish_id INT, PRIMARY KEY (meal_id, dish_id), FOREIGN KEY (meal_id) REFERENCES meals(id), FOREIGN KEY (dish_id) REFERENCES dishes(id));
